import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.io.RandomAccessFile;

public class DLG extends JDialog {
    private JPanel contentPane;
    private JButton JBGoida;
    private JTextField EFirst;
    private JTextField ESecond;

    RandomAccessFile RafFirst;

    RandomAccessFile RafSecond;

    String FileNameFirst;

    String FileNameSecond;

    byte UFirst[];

    byte USecond[];
    int i, NFirst, NSecond;

    public DLG() {
        setContentPane(contentPane);
        setModal(true);

        JBGoida.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {

                FileNameFirst = EFirst.getText();

                FileNameSecond = ESecond.getText();

                try {
                    RafFirst = new RandomAccessFile(FileNameFirst, "rw");

                    RafSecond = new RandomAccessFile(FileNameSecond, "rw");

                    NFirst = (int) RafFirst.length();

                    NSecond = (int) RafSecond.length();

                    if (NFirst != NSecond) {
                        JOptionPane.showMessageDialog(null, "Размеры Файлов Не Равны", "Message", JOptionPane.PLAIN_MESSAGE);

                        RafFirst.close();

                        RafSecond.close();

                        return;
                    }

                    UFirst = new byte[NFirst];

                    USecond = new byte[NSecond];

                    RafFirst.read(UFirst, 0, NFirst);

                    RafSecond.read(USecond, 0, NSecond);

                    for (i = 0; i <= NFirst - 1; i++)

                        if (UFirst[i] != USecond[i]) {
                            JOptionPane.showMessageDialog(null, "Файлы Не Равны", "Message", JOptionPane.PLAIN_MESSAGE);

                            RafFirst.close();

                            RafSecond.close();

                            return;
                        }

                    if (i == NFirst) {
                        JOptionPane.showMessageDialog(null, "Файлы Равны", "Message", JOptionPane.PLAIN_MESSAGE);

                        RafFirst.close();

                        RafSecond.close();

                        return;
                    }

                } catch (Exception exc) {
                }
            }
        });
    }

    public static void main(String[] args) {
        DLG dialog = new DLG();
        dialog.pack();
        dialog.setVisible(true);
        System.exit(0);
    }
