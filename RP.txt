1. Придумываем БД (1 сущность, 5 столбцов), заполняем таблицу, пофиг.

2. Перейти в Visual Studio. Создание проекта -> Веб-приложение ASP.NET Core.

3. ПКМ по решению, создаем папку Models, а в ней класс Product.cs (или др. название по теме).

4. Product.cs:
using System.ComponentModel.DataAnnotations.Schema;
using System.ComponentModel.DataAnnotations;
using Microsoft.EntityFrameworkCore;

namespace НАЗВАНИЕ_ПРОЕКТА_В_ВИЗУАЛКЕ.Models
{
    public class Product
    {
        [Key]
        public int Id { get; set; }

        [Required(ErrorMessage = "Название обязательно")]
        [StringLength(100, ErrorMessage = "Название товара не должно превышать 100 символов")]
        [Display(Name = "Название продукта")]
        public string Name { get; set; }

        [Required(ErrorMessage = "Цена продукта обязательна")]
        [Range(0.01, 10000, ErrorMessage = "Цена должна быть между 0.01 и 10000")]
        [Display(Name = "Цена")]
        [Precision(10,2)]
        [Column(TypeName = "decimal(18,2)")]
        public decimal Price { get; set; }

        [Required(ErrorMessage = "Описание товара обязательно")]
        [StringLength(1000, ErrorMessage = "Описание товара не должно превышать 1000 символов")]
        [Display(Name = "Описание продукта")]
        public string Description { get; set; }

        [Required(ErrorMessage = "Введите кол-во товара на складе")]
        [Range(1, 1000000, ErrorMessage = "Остаток на складе должен быть от 1шт. до 1000000шт.")]
        [Display(Name = "Остаток на складе")]
        public int StockQuantity { get; set; }
    }
}

6. Models -> Создать класс ApplicationDbContext.cs.

7. ApplicationDbContext.cs:
using Microsoft.EntityFrameworkCore;

namespace НАЗВАНИЕ_ПРОЕКТА_В_ВИЗУАЛКЕ.Models
{
    public class ApplicationDbContext : DbContext
    {
        public ApplicationDbContext(DbContextOptions<ApplicationDbContext> options)
            : base(options)
        {
        }

        public DbSet<Product> Products { get; set; }

        protected override void OnModelCreating(ModelBuilder modelBuilder)
        {
            base.OnModelCreating(modelBuilder);

            modelBuilder.Entity<Product>(entity =>

            entity.Property(p => p.Price)
                  .HasPrecision(10,2)
            );
        }
    }
}
